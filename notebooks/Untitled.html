import re
import itertools
def cleanup_list(chunks):
    """remove empty elements from the list"""
    chunks = map(lambda x: x.strip(), chunks)
    chunks = filter(lambda x: x, chunks)
    return list(chunks)

def header_level(header):
    return len(header.split(' ')[0])

def prepare_math_equations(text_md):
    """Make sure block math equations are not paragraphs on their own."""
    block_math_pattern = r'(\\\[[\s\S]*?\\\]|\$\$[\s\S]*?\$\$)'
    chunks = re.split(block_math_pattern, text_md, flags=re.MULTILINE)
    chunks = cleanup_list(chunks)
    updated_text_md = '\n'.join(chunks)
    return updated_text_md
def parse_markdown(text_md):
    """
    Assume the first header is at level 1, and it is the title of the paper.
    """
    header_pattern = r'^(#{1,6} .*?)$'
    sections = re.split(header_pattern, text_md, flags=re.MULTILINE)
    sections = cleanup_list(sections)
    doc_title = sections[0][2:]

    start_idx = 1
    for i, section in enumerate(sections[1:], start=1):
        if section.startswith('#'):
            start_idx = i
            break

    paragraphs = map(lambda x: x.split('\n\n'), sections[start_idx:])
    paragraphs = itertools.chain.from_iterable(paragraphs)
    paragraphs = cleanup_list(paragraphs)
    return doc_title, paragraphs

def to_text_nodes(paragraphs):
    headers = []
    chunks = []
    for i, paragraph in enumerate(paragraphs):
        if paragraph.startswith('#'):
            # handle headers list
            if not headers: 
                # this is the first section
                headers.append(paragraph)
            else:
                # construct headers list for the current paragraph
                curr_level = header_level(paragraph)
                
                curr_headers = []
                for header in headers:
                    if header_level(header) < curr_level:
                        curr_headers.append(header)
                    else:
                        break
                curr_headers.append(paragraph)
                
                headers = curr_headers
        else:
            curr_chunk = {'text': paragraph, 'headers': headers}
            chunks.append(curr_chunk)

    return chunks
